SlideOutPanel Examples

Slide From

							$('#slide-out-panel').SlideOutPanel();

							$('#slide-out-panel').SlideOutPanel({
								   slideFrom: 'left',
							});

              $('#slide-out-panel').SlideOutPanel({
                   slideFrom: 'top',
              });

              $('#slide-out-panel').SlideOutPanel({
                   slideFrom: 'bottom',
              });

Body Push

              $('#slide-out-panel').SlideOutPanel({
                   bodyPush: true,
              });

              $('#slide-out-panel').SlideOutPanel({
                   bodyPush: true,
                   slideFrom: 'left',
              });

Screen Close

              $('#slide-out-panel').SlideOutPanel({
                   screenClose: false,
              });

Show Screen

              $('#slide-out-panel').SlideOutPanel({
                   showScreen: false,
              });

Transitions

              $('#slide-out-panel').SlideOutPanel({
                   transitionDuration: '5s',
              });
						

              $('#slide-out-panel').SlideOutPanel({
                   transition: 'ease-in-out',
              });

Offset Top

              $('#slide-out-panel').SlideOutPanel({
                   offsetTop: '50px',
              });

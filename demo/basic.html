SlideOutPanel Examples

Slide from directions

              $('#slide-out-panel').SlideOutPanel({
                   slideFrom: 'top',
              });

              $('#slide-out-panel').SlideOutPanel({
                   slideFrom: 'right',
              });

              $('#slide-out-panel').SlideOutPanel({
                   slideFrom: 'bottom',
              });

              $('#slide-out-panel').SlideOutPanel({
                   slideFrom: 'left',
              });

Body Push

              $('#slide-out-panel').SlideOutPanel({
                   bodyPush: true,
                   slideFrom: 'right',
              });

              $('#slide-out-panel').SlideOutPanel({
                   bodyPush: true,
                   slideFrom: 'left',
              });

Offset Top

              $('#slide-out-panel').SlideOutPanel({
                   offsetTop: '50px',
              });
